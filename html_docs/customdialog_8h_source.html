6 // This file contains a special class, "CustomDialog", for generating
7 // a modal dialog box with multiple form elements/inputs including:
8 // labels, checkboxes, line edit boxes, number edit boxes, combo boxes,
9 // radio buttons, spin boxes (for ints or floats), group boxes (to group elements)
10 // and even "pick color" buttons (letting you chose a color).
11 //
12 // NOTES:
13 //
14 // > Each form element is created by a single function only, includes an optional tooltip,
15 // and a pointer to the variable you wish to modify.
16 //
17 // > The value of the variable is updated ONLY if and when hits "Ok", and will change only
18 // if the user has changed it from the original/default/starting value in that variable.
19 //
20 // > This class if fantastic for making modal dialogs (i.e. dialogs where you can't click
21 // anything else until you click okay or cancel), but isn't designed for times when you
22 // want/need the parent dialog to dymaically update as the user changes values
23 // (i.e. interactive changes).
24 //
25 // > This method of generating dialog elements on-the-fly by using a vectors of structs is
26 // obviously less versatile than creating a seperate class which extend QDialog for EACH
27 // dialog you need. The huge advantage, however, is you can can greatly reduce your code
28 // and avoid the tedium of creating a seperate .h and .cpp file everytime you wants to
29 // prompt the user for multiple pieces of info in a single modal dialog!
30 //
31 // > This file also includes several "convinience" functions such as "MsgBoxYesNo" and
32 // "setBold" for quickly displaying dialogs/returning results and modifying strings.
33 // These convinience functions are at the bottom of this file (and defined in the .cpp).
34 //
35 // -----------------
36 //
37 // An example of how to use this dialog follows:
38 //
39 // void GetDetailsFromUser()
40 // {
41 // string name = "First Last";
42 // bool student = true;
43 // int age = 20;
44 // int sportIdx = 1;
45 //
46 // CustomDialog d("Registration", this);
47 // d.addLabel ("Please enter your details below ...");
48 // d.addLineEdit ("name: ", &name, "No middle name!");
49 // d.addCheckBox ("current student", &student);
50 // d.addSpinBox ("your age: ", 1, 120, &age, 1);
51 // d.addComboBox ("sport: ", "tennis|soccer|cricket", &sportIdx);
52 //
53 // d.exec(); // Execution stops here until user closes dialog
54 //
55 // if(d.wasCancelled()) {
56 // cout << "You have cancelled your registration" << endl;
57 // return;
58 // }
59 //
60 // cout << "Dear " << name << " thank you for registering..." << endl;
61 //
62 // ... CODE ...
63 // }
64 //
65 // -----------------
66 //
67 // And produces a dialog which looks like this:

69 // +----------------------------------------+
